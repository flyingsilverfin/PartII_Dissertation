\makeatletter
\newcommand\resetstackedplots{
	\makeatletter
	\pgfplots@stacked@isfirstplottrue
	\makeatother
	\addplot [forget plot,draw=none] coordinates{(1,0) (2,0) (5,0) (10,0) (50,0)};
}

\makeatother
